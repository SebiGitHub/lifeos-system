@Composable
fun RankingScreen(
    vm: RankingViewModel,
    modifier: Modifier = Modifier
) {
    val state by vm.state.collectAsState()
    val ctx = LocalContext.current

    var hasPerm by remember { mutableStateOf(UsageAccess.hasUsageAccess(ctx)) }

    // Refresca cuando vuelves a la app (por si concediste permiso)
    val lifecycleOwner = androidx.lifecycle.compose.LocalLifecycleOwner.current

    DisposableEffect(lifecycleOwner) {
        val obs = LifecycleEventObserver { _, event ->
            if (event == Lifecycle.Event.ON_RESUME) {
                hasPerm = UsageAccess.hasUsageAccess(ctx)
                if (hasPerm) vm.refresh()
            }
        }

        val lifecycle = lifecycleOwner.lifecycle
        lifecycle.addObserver(obs)
        onDispose { lifecycle.removeObserver(obs) }
    }


    Column(modifier = modifier.padding(16.dp)) {
        ScreenTitle("Ranking")
        Spacer(Modifier.height(12.dp))

        if (!hasPerm) {
            Card(
                colors = CardDefaults.cardColors(
                    containerColor = MaterialTheme.colorScheme.errorContainer
                ),
                modifier = Modifier.fillMaxWidth()
            ) {
                Column(Modifier.padding(14.dp)) {
                    Text(
                        text = "Falta permiso de uso para calcular el ranking.",
                        fontWeight = FontWeight.SemiBold,
                        color = MaterialTheme.colorScheme.onErrorContainer
                    )
                    Spacer(Modifier.height(8.dp))
                    Text(
                        text = "Toca aquí para habilitarlo.",
                        color = MaterialTheme.colorScheme.onErrorContainer,
                        modifier = Modifier
                            .clip(RoundedCornerShape(10.dp))
                            .background(MaterialTheme.colorScheme.onErrorContainer.copy(alpha = 0.12f))
                            .clickable { UsageAccess.openUsageAccessSettings(ctx) }
                            .padding(horizontal = 10.dp, vertical = 8.dp)
                            .fillMaxWidth()
                    )
                }
            }
            return
        }

        val topDay = remember(state.topDay) { state.topDay.map(::toRankItem) }
        val topYear = remember(state.topYear) { state.topYear.map(::toRankItem) }

        SectionTitle("Top de hoy")
        Podium(top3 = topDay)

        Spacer(Modifier.height(18.dp))

        SectionTitle("Top del año")
        Podium(top3 = topYear)
    }
}

private data class RankItem(
    val packageName: String,
    val label: String,
    val totalMs: Long
)

private fun toRankItem(row: AppUsageAgg): RankItem = RankItem(
    packageName = row.packageName,
    label = row.label ?: row.packageName,
    totalMs = row.totalMs
)

@Composable
private fun Podium(top3: List<RankItem>) {
    val items = remember(top3) {
        val list = top3.take(3).toMutableList()
        while (list.size < 3) list.add(RankItem("", "", 0L))
        list
    }

    val pulse = rememberInfiniteTransition(label = "pulse")
        .animateFloat(
            initialValue = 0.98f,
            targetValue = 1.02f,
            animationSpec = infiniteRepeatable(
                animation = tween(900),
                repeatMode = RepeatMode.Reverse
            ),
            label = "pulseFloat"
        ).value

    // Podio tipo "escalón" (3-1-2) con trofeo arriba en el centro
    Box(
        modifier = Modifier
            .fillMaxWidth()
            .padding(top = 10.dp)
    ) {
        Row(
            modifier = Modifier
                .fillMaxWidth()
                .padding(top = 24.dp),
            horizontalArrangement = Arrangement.spacedBy(0.dp),
            verticalAlignment = Alignment.Bottom
        ) {
            PodiumStep(
                position = 3,
                item = items.getOrNull(2) ?: RankItem("", "", 0L),
                stepHeight = 86.dp,
                modifier = Modifier.weight(1f),
                scale = 1f
            )
            PodiumStep(
                position = 1,
                item = items.getOrNull(0) ?: RankItem("", "", 0L),
                stepHeight = 118.dp,
                modifier = Modifier.weight(1.15f),
                scale = pulse
            )
            PodiumStep(
                position = 2,
                item = items.getOrNull(1) ?: RankItem("", "", 0L),
                stepHeight = 98.dp,
                modifier = Modifier.weight(1f),
                scale = 1f
            )
        }

        Icon(
            imageVector = Icons.Filled.EmojiEvents,
            contentDescription = null,
            tint = MaterialTheme.colorScheme.tertiary,
            modifier = Modifier
                .align(Alignment.TopCenter)
                .size(42.dp)
        )
    }
}

@Composable
private fun PodiumStep(
    position: Int,
    item: RankItem,
    stepHeight: androidx.compose.ui.unit.Dp,
    modifier: Modifier = Modifier,
    scale: Float = 1f
) {
    val (bg, fg) = when (position) {
        1 -> MaterialTheme.colorScheme.primary to MaterialTheme.colorScheme.onPrimary
        2 -> MaterialTheme.colorScheme.tertiary to MaterialTheme.colorScheme.onTertiary
        else -> MaterialTheme.colorScheme.secondary to MaterialTheme.colorScheme.onSecondary
    }
    val cap = MaterialTheme.colorScheme.surfaceVariant.copy(alpha = 0.75f)

    Column(
        modifier = modifier,
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Box(
            modifier = Modifier
                .fillMaxWidth()
                .height(stepHeight)
                .padding(horizontal = 2.dp)
                .graphicsLayer {
                    scaleX = scale
                    scaleY = scale
                }
                .clip(RoundedCornerShape(14.dp))
                .background(bg)
        ) {
            // Tapa superior (gris) como la imagen
            Box(
                modifier = Modifier
                    .fillMaxWidth()
                    .height(12.dp)
                    .background(cap),
                contentAlignment = Alignment.Center
            ) {
                if (item.packageName.isNotBlank()) {
                    AppIcon(packageName = item.packageName, size = 18.dp)
                }
            }

            // Número centrado
            Text(
                text = position.toString(),
                color = fg,
                style = MaterialTheme.typography.titleLarge,
                fontWeight = FontWeight.ExtraBold,
                modifier = Modifier.align(Alignment.Center)
            )
        }

        Spacer(Modifier.height(8.dp))

        // Abajo del podio: TIEMPO (no el nombre)
        Text(
            text = if (item.totalMs > 0L) TimeFmt.msToHuman(item.totalMs) else "--",
            style = MaterialTheme.typography.bodySmall,
            color = MaterialTheme.colorScheme.onSurface.copy(alpha = 0.75f),
            modifier = Modifier.fillMaxWidth(),
            textAlign = TextAlign.Center
        )
    }
}
